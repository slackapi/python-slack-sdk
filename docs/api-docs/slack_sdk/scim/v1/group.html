    def __init__(
        self,
        *,

        id: Union[Optional[str], DefaultArg] = NotGiven,

        meta: Union[Optional[GroupMeta], DefaultArg] = NotGiven,

        **kwargs,
    ) -> None:
        self.display_name = display_name

        self.schemas = schemas
